251/// Encodings supported by Parquet.
252///
253/// Not all encodings are valid for all types. These enums are also used to specify the
254/// encoding of definition and repetition levels.
255///
256/// By default this crate uses [Encoding::PLAIN], [Encoding::RLE], and [Encoding::RLE_DICTIONARY].
257/// These provide very good encode and decode performance, whilst yielding reasonable storage
258/// efficiency and being supported by all major parquet readers.
259///
260/// The delta encodings are also supported and will be used if a newer [WriterVersion] is
261/// configured, however, it should be noted that these sacrifice encode and decode performance for
262/// improved storage efficiency. This performance regression is particularly pronounced in the case
263/// of record skipping as occurs during predicate push-down. It is recommended users assess the
264/// performance impact when evaluating these encodings.
265///
266/// [WriterVersion]: crate::file::properties::WriterVersion
267#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Ord, PartialOrd)]
268#[allow(non_camel_case_types)]
269pub enum Encoding {
270    /// Default byte encoding.
271    /// - BOOLEAN - 1 bit per value, 0 is false; 1 is true.
272    /// - INT32 - 4 bytes per value, stored as little-endian.
273    /// - INT64 - 8 bytes per value, stored as little-endian.
274    /// - FLOAT - 4 bytes per value, stored as little-endian.
275    /// - DOUBLE - 8 bytes per value, stored as little-endian.
276    /// - BYTE_ARRAY - 4 byte length stored as little endian, followed by bytes.
277    /// - FIXED_LEN_BYTE_ARRAY - just the bytes are stored.
278    PLAIN,
279
280    /// **Deprecated** dictionary encoding.
281    ///
282    /// The values in the dictionary are encoded using PLAIN encoding.
283    /// Since it is deprecated, RLE_DICTIONARY encoding is used for a data page, and
284    /// PLAIN encoding is used for dictionary page.
285    PLAIN_DICTIONARY,
286
287    /// Group packed run length encoding.
288    ///
289    /// Usable for definition/repetition levels encoding and boolean values.
290    RLE,
291
292    /// **Deprecated** Bit-packed encoding.
293    ///
294    /// This can only be used if the data has a known max width.
295    /// Usable for definition/repetition levels encoding.
296    ///
297    /// There are compatibility issues with files using this encoding.
298    /// The parquet standard specifies the bits to be packed starting from the
299    /// most-significant bit, several implementations do not follow this bit order.
300    /// Several other implementations also have issues reading this encoding
301    /// because of incorrect assumptions about the length of the encoded data.
302    ///
303    /// The RLE/bit-packing hybrid is more cpu and memory efficient and should be used instead.
304    #[deprecated(
305        since = "51.0.0",
306        note = "Please see documentation for compatibility issues and use the RLE/bit-packing hybrid encoding instead"
307    )]
308    BIT_PACKED,
309
310    /// Delta encoding for integers, either INT32 or INT64.
311    ///
312    /// Works best on sorted data.
313    DELTA_BINARY_PACKED,
314
315    /// Encoding for byte arrays to separate the length values and the data.
316    ///
317    /// The lengths are encoded using DELTA_BINARY_PACKED encoding.
318    DELTA_LENGTH_BYTE_ARRAY,
319
320    /// Incremental encoding for byte arrays.
321    ///
322    /// Prefix lengths are encoded using DELTA_BINARY_PACKED encoding.
323    /// Suffixes are stored using DELTA_LENGTH_BYTE_ARRAY encoding.
324    DELTA_BYTE_ARRAY,
325
326    /// Dictionary encoding.
327    ///
328    /// The ids are encoded using the RLE encoding.
329    RLE_DICTIONARY,
330
331    /// Encoding for fixed-width data.
332    ///
333    /// K byte-streams are created where K is the size in bytes of the data type.
334    /// The individual bytes of a value are scattered to the corresponding stream and
335    /// the streams are concatenated.
336    /// This itself does not reduce the size of the data but can lead to better compression
337    /// afterwards. Note that the use of this encoding with FIXED_LEN_BYTE_ARRAY(N) data may
338    /// perform poorly for large values of N.
339    BYTE_STREAM_SPLIT,
340}
341
342impl FromStr for Encoding {
343    type Err = ParquetError;
344
345    fn from_str(s: &str) -> Result<Self, Self::Err> {
346        match s {
347            "PLAIN" | "plain" => Ok(Encoding::PLAIN),
348            "PLAIN_DICTIONARY" | "plain_dictionary" => Ok(Encoding::PLAIN_DICTIONARY),
349            "RLE" | "rle" => Ok(Encoding::RLE),
350            #[allow(deprecated)]
351            "BIT_PACKED" | "bit_packed" => Ok(Encoding::BIT_PACKED),
352            "DELTA_BINARY_PACKED" | "delta_binary_packed" => Ok(Encoding::DELTA_BINARY_PACKED),
353            "DELTA_LENGTH_BYTE_ARRAY" | "delta_length_byte_array" => {
354                Ok(Encoding::DELTA_LENGTH_BYTE_ARRAY)
355            }
356            "DELTA_BYTE_ARRAY" | "delta_byte_array" => Ok(Encoding::DELTA_BYTE_ARRAY),
357            "RLE_DICTIONARY" | "rle_dictionary" => Ok(Encoding::RLE_DICTIONARY),
358            "BYTE_STREAM_SPLIT" | "byte_stream_split" => Ok(Encoding::BYTE_STREAM_SPLIT),
359            _ => Err(general_err!("unknown encoding: {}", s)),
360        }
361    }
362}

367/// Supported block compression algorithms.
368///
369/// Block compression can yield non-trivial improvements to storage efficiency at the expense
370/// of potentially significantly worse encode and decode performance. Many applications,
371/// especially those making use of high-throughput and low-cost commodity object storage,
372/// may find storage efficiency less important than decode throughput, and therefore may
373/// wish to not make use of block compression.
374///
375/// The writers in this crate default to no block compression for this reason.
376///
377/// Applications that do still wish to use block compression, will find [`Compression::ZSTD`]
378/// to provide a good balance of compression, performance, and ecosystem support. Alternatively,
379/// [`Compression::LZ4_RAW`] provides much faster decompression speeds, at the cost of typically
380/// worse compression ratios. However, it is not as widely supported by the ecosystem, with the
381/// Hadoop ecosystem historically favoring the non-standard and now deprecated [`Compression::LZ4`].
382#[derive(Debug, Clone, Copy, PartialEq, Eq)]
383#[allow(non_camel_case_types)]
384pub enum Compression {
385    /// No compression.
386    UNCOMPRESSED,
387    /// [Snappy compression](https://en.wikipedia.org/wiki/Snappy_(compression))
388    SNAPPY,
389    /// [Gzip compression](https://www.ietf.org/rfc/rfc1952.txt)
390    GZIP(GzipLevel),
391    /// [LZO compression](https://en.wikipedia.org/wiki/Lempel%E2%80%93Ziv%E2%80%93Oberhumer)
392    LZO,
393    /// [Brotli compression](https://datatracker.ietf.org/doc/html/rfc7932)
394    BROTLI(BrotliLevel),
395    /// [LZ4 compression](https://lz4.org/), [(deprecated)](https://issues.apache.org/jira/browse/PARQUET-2032)
396    LZ4,
397    /// [ZSTD compression](https://datatracker.ietf.org/doc/html/rfc8878)
398    ZSTD(ZstdLevel),
399    /// [LZ4 compression](https://lz4.org/).
400    LZ4_RAW,
401}
402
403impl Compression {
404    /// Returns the codec type of this compression setting as a string, without the compression
405    /// level.
406    pub(crate) fn codec_to_string(self) -> String {
407        format!("{:?}", self).split('(').next().unwrap().to_owned()
408    }
409}
410
411fn split_compression_string(str_setting: &str) -> Result<(&str, Option<u32>), ParquetError> {
412    let split_setting = str_setting.split_once('(');
413
414    match split_setting {
415        Some((codec, level_str)) => {
416            let level = &level_str[..level_str.len() - 1]
417                .parse::<u32>()
418                .map_err(|_| {
419                    ParquetError::General(format!("invalid compression level: {}", level_str))
420                })?;
421            Ok((codec, Some(*level)))
422        }
423        None => Ok((str_setting, None)),
424    }
425}
426
427fn check_level_is_none(level: &Option<u32>) -> Result<(), ParquetError> {
428    if level.is_some() {
429        return Err(ParquetError::General(
430            "compression level is not supported".to_string(),
431        ));
432    }
433
434    Ok(())
435}
436
437fn require_level(codec: &str, level: Option<u32>) -> Result<u32, ParquetError> {
438    level.ok_or(ParquetError::General(format!(
439        "{} requires a compression level",
440        codec
441    )))
442}
443
444impl FromStr for Compression {
445    type Err = ParquetError;
446
447    fn from_str(s: &str) -> std::result::Result<Self, Self::Err> {
448        let (codec, level) = split_compression_string(s)?;
449
450        let c = match codec {
451            "UNCOMPRESSED" | "uncompressed" => {
452                check_level_is_none(&level)?;
453                Compression::UNCOMPRESSED
454            }
455            "SNAPPY" | "snappy" => {
456                check_level_is_none(&level)?;
457                Compression::SNAPPY
458            }
459            "GZIP" | "gzip" => {
460                let level = require_level(codec, level)?;
461                Compression::GZIP(GzipLevel::try_new(level)?)
462            }
463            "LZO" | "lzo" => {
464                check_level_is_none(&level)?;
465                Compression::LZO
466            }
467            "BROTLI" | "brotli" => {
468                let level = require_level(codec, level)?;
469                Compression::BROTLI(BrotliLevel::try_new(level)?)
470            }
471            "LZ4" | "lz4" => {
472                check_level_is_none(&level)?;
473                Compression::LZ4
474            }
475            "ZSTD" | "zstd" => {
476                let level = require_level(codec, level)?;
477                Compression::ZSTD(ZstdLevel::try_new(level as i32)?)
478            }
479            "LZ4_RAW" | "lz4_raw" => {
480                check_level_is_none(&level)?;
481                Compression::LZ4_RAW
482            }
483            _ => {
484                return Err(ParquetError::General(format!(
485                    "unsupport compression {codec}"
486                )));
487            }
488        };
489
490        Ok(c)
491    }
492}

515/// Sort order for page and column statistics.
516///
517/// Types are associated with sort orders and column stats are aggregated using a sort
518/// order, and a sort order should be considered when comparing values with statistics
519/// min/max.
520///
521/// See reference in
522/// <https://github.com/apache/arrow/blob/main/cpp/src/parquet/types.h>
523#[derive(Debug, Clone, Copy, PartialEq, Eq)]
524#[allow(non_camel_case_types)]
525pub enum SortOrder {
526    /// Signed (either value or legacy byte-wise) comparison.
527    SIGNED,
528    /// Unsigned (depending on physical type either value or byte-wise) comparison.
529    UNSIGNED,
530    /// Comparison is undefined.
531    UNDEFINED,
532}
533
534impl SortOrder {
535    /// Returns true if this is [`Self::SIGNED`]
536    pub fn is_signed(&self) -> bool {
537        matches!(self, Self::SIGNED)
538    }
539}
540
541/// Column order that specifies what method was used to aggregate min/max values for
542/// statistics.
543///
544/// If column order is undefined, then it is the legacy behaviour and all values should
545/// be compared as signed values/bytes.
546#[derive(Debug, Clone, Copy, PartialEq, Eq)]
547#[allow(non_camel_case_types)]
548pub enum ColumnOrder {
549    /// Column uses the order defined by its logical or physical type
550    /// (if there is no logical type), parquet-format 2.4.0+.
551    TYPE_DEFINED_ORDER(SortOrder),
552    /// Undefined column order, means legacy behaviour before parquet-format 2.4.0.
553    /// Sort order is always SIGNED.
554    UNDEFINED,
555}
556
557impl ColumnOrder {
558    /// Returns sort order for a physical/logical type.
559    pub fn get_sort_order(
560        logical_type: Option<LogicalType>,
561        converted_type: ConvertedType,
562        physical_type: Type,
563    ) -> SortOrder {
564        // TODO: Should this take converted and logical type, for compatibility?
565        match logical_type {
566            Some(logical) => match logical {
567                LogicalType::String | LogicalType::Enum | LogicalType::Json | LogicalType::Bson => {
568                    SortOrder::UNSIGNED
569                }
570                LogicalType::Integer { is_signed, .. } => match is_signed {
571                    true => SortOrder::SIGNED,
572                    false => SortOrder::UNSIGNED,
573                },
574                LogicalType::Map | LogicalType::List => SortOrder::UNDEFINED,
575                LogicalType::Decimal { .. } => SortOrder::SIGNED,
576                LogicalType::Date => SortOrder::SIGNED,
577                LogicalType::Time { .. } => SortOrder::SIGNED,
578                LogicalType::Timestamp { .. } => SortOrder::SIGNED,
579                LogicalType::Unknown => SortOrder::UNDEFINED,
580                LogicalType::Uuid => SortOrder::UNSIGNED,
581                LogicalType::Float16 => SortOrder::SIGNED,
582            },
583            // Fall back to converted type
584            None => Self::get_converted_sort_order(converted_type, physical_type),
585        }
586    }
587
588    fn get_converted_sort_order(converted_type: ConvertedType, physical_type: Type) -> SortOrder {
589        match converted_type {
590            // Unsigned byte-wise comparison.
591            ConvertedType::UTF8
592            | ConvertedType::JSON
593            | ConvertedType::BSON
594            | ConvertedType::ENUM => SortOrder::UNSIGNED,
595
596            ConvertedType::INT_8
597            | ConvertedType::INT_16
598            | ConvertedType::INT_32
599            | ConvertedType::INT_64 => SortOrder::SIGNED,
600
601            ConvertedType::UINT_8
602            | ConvertedType::UINT_16
603            | ConvertedType::UINT_32
604            | ConvertedType::UINT_64 => SortOrder::UNSIGNED,
605
606            // Signed comparison of the represented value.
607            ConvertedType::DECIMAL => SortOrder::SIGNED,
608
609            ConvertedType::DATE => SortOrder::SIGNED,
610
611            ConvertedType::TIME_MILLIS
612            | ConvertedType::TIME_MICROS
613            | ConvertedType::TIMESTAMP_MILLIS
614            | ConvertedType::TIMESTAMP_MICROS => SortOrder::SIGNED,
615
616            ConvertedType::INTERVAL => SortOrder::UNDEFINED,
617
618            ConvertedType::LIST | ConvertedType::MAP | ConvertedType::MAP_KEY_VALUE => {
619                SortOrder::UNDEFINED
620            }
621
622            // Fall back to physical type.
623            ConvertedType::NONE => Self::get_default_sort_order(physical_type),
624        }
625    }
626
627    /// Returns default sort order based on physical type.
628    fn get_default_sort_order(physical_type: Type) -> SortOrder {
629        match physical_type {
630            // Order: false, true
631            Type::BOOLEAN => SortOrder::UNSIGNED,
632            Type::INT32 | Type::INT64 => SortOrder::SIGNED,
633            Type::INT96 => SortOrder::UNDEFINED,
634            // Notes to remember when comparing float/double values:
635            // If the min is a NaN, it should be ignored.
636            // If the max is a NaN, it should be ignored.
637            // If the min is +0, the row group may contain -0 values as well.
638            // If the max is -0, the row group may contain +0 values as well.
639            // When looking for NaN values, min and max should be ignored.
640            Type::FLOAT | Type::DOUBLE => SortOrder::SIGNED,
641            // Unsigned byte-wise comparison
642            Type::BYTE_ARRAY | Type::FIXED_LEN_BYTE_ARRAY => SortOrder::UNSIGNED,
643        }
644    }
645
646    /// Returns sort order associated with this column order.
647    pub fn sort_order(&self) -> SortOrder {
648        match *self {
649            ColumnOrder::TYPE_DEFINED_ORDER(order) => order,
650            ColumnOrder::UNDEFINED => SortOrder::SIGNED,
651        }
652    }
653}

890// LogicalType <=> ConvertedType conversion
891
892// Note: To prevent type loss when converting from ConvertedType to LogicalType,
893// the conversion from ConvertedType -> LogicalType is not implemented.
894// Such type loss includes:
895// - Not knowing the decimal scale and precision of ConvertedType
896// - Time and timestamp nanosecond precision, that is not supported in ConvertedType.
897
898impl From<Option<LogicalType>> for ConvertedType {
899    fn from(value: Option<LogicalType>) -> Self {
900        match value {
901            Some(value) => match value {
902                LogicalType::String => ConvertedType::UTF8,
903                LogicalType::Map => ConvertedType::MAP,
904                LogicalType::List => ConvertedType::LIST,
905                LogicalType::Enum => ConvertedType::ENUM,
906                LogicalType::Decimal { .. } => ConvertedType::DECIMAL,
907                LogicalType::Date => ConvertedType::DATE,
908                LogicalType::Time { unit, .. } => match unit {
909                    TimeUnit::MILLIS(_) => ConvertedType::TIME_MILLIS,
910                    TimeUnit::MICROS(_) => ConvertedType::TIME_MICROS,
911                    TimeUnit::NANOS(_) => ConvertedType::NONE,
912                },
913                LogicalType::Timestamp { unit, .. } => match unit {
914                    TimeUnit::MILLIS(_) => ConvertedType::TIMESTAMP_MILLIS,
915                    TimeUnit::MICROS(_) => ConvertedType::TIMESTAMP_MICROS,
916                    TimeUnit::NANOS(_) => ConvertedType::NONE,
917                },
918                LogicalType::Integer {
919                    bit_width,
920                    is_signed,
921                } => match (bit_width, is_signed) {
922                    (8, true) => ConvertedType::INT_8,
923                    (16, true) => ConvertedType::INT_16,
924                    (32, true) => ConvertedType::INT_32,
925                    (64, true) => ConvertedType::INT_64,
926                    (8, false) => ConvertedType::UINT_8,
927                    (16, false) => ConvertedType::UINT_16,
928                    (32, false) => ConvertedType::UINT_32,
929                    (64, false) => ConvertedType::UINT_64,
930                    t => panic!("Integer type {t:?} is not supported"),
931                },
932                LogicalType::Json => ConvertedType::JSON,
933                LogicalType::Bson => ConvertedType::BSON,
934                LogicalType::Uuid | LogicalType::Float16 | LogicalType::Unknown => {
935                    ConvertedType::NONE
936                }
937            },
938            None => ConvertedType::NONE,
939        }
940    }
